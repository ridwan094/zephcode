CSS Snippet for Tooltips


.zc-tooltip {
    position: relative;
    display: inline-block;
}

.zc-tooltip .zc-tooltip-text {
    visibility: hidden;
    background-color: #333;
    color: #fff;
    text-align: center;
    border-radius: 4px;
    padding: 5px;
    position: absolute;
    z-index: 1;
    opacity: 0;
    transition: opacity 0.3s;
}

.zc-tooltip:hover .zc-tooltip-text {
    visibility: visible;
    opacity: 1;
}
