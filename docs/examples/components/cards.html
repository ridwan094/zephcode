CSS Snippet for Cards

            
            /* Default Card */
            .zc-card {
                border: 1px solid #ced4da;
                border-radius: 0.25rem;
                box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1);
                padding: 1rem;
                background-color: #fff;
            }
            
            /* Small Card */
            .zc-card.small {
                width: 18rem;
            }
            
            /* Gradient Card */
            .zc-card.gradient {
                background: linear-gradient(to right, #007bff, #28a745);
                color: #fff;
            }
            
            /* Media Card */
            .zc-card.media {
                position: relative;
            }
            
            .zc-card.media img {
                width: 100%;
                border-radius: 0.25rem 0.25rem 0 0;
            }
            
            .zc-card-media-overlay {
                position: absolute;
                bottom: 0;
                left: 0;
                right: 0;
                background: rgba(0, 0, 0, 0.5);
                color: #fff;
                padding: 0.5rem;
                text-align: center;
            }
            
            /* Hover Effect for Floating Card */
            .zc-card.floating:hover {
                transform: translateY(-10px) scale(1.02);
                box-shadow: 0 8px 16px rgba(0, 0, 0, 0.2);
            }
