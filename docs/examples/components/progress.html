CSS Snippet for Progress Bar

            
            .zc-progress {
                width: 100%;
                height: 1rem;
                background-color: #e9ecef;
                border-radius: 0.25rem;
                overflow: hidden;
            }
            
            .zc-progress-bar {
                height: 100%;
                transition: width 0.6s ease;
            }
            
            .zc-progress-bar.striped {
                background-image: linear-gradient(45deg, rgba(255,255,255,0.15) 25%, transparent 25%, transparent 50%, rgba(255,255,255,0.15) 50%);
                background-size: 1rem 1rem;
            }
            
            .zc-progress-bar.animated {
                animation: progress-bar-stripes 1s linear infinite;
            }
            
            @keyframes progress-bar-stripes {
                from { background-position: 1rem 0; }
                to { background-position: 0 0; }
            }
            
            .zc-progress-sm {
                height: 0.75rem;
            }
            
            .zc-progress-lg {
                height: 1.5rem;
            }
            
            .zc-progress-gradient {
                background: linear-gradient(to right, #007bff, #17a2b8);
            }
