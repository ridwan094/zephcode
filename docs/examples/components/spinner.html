CSS Snippet for Spinner

            
            .zc-spinner {
                width: 2rem;
                height: 2rem;
                border: 0.25rem solid #e9ecef;
                border-top: 0.25rem solid #007bff;
                border-radius: 50%;
                animation: spin 1s linear infinite;
            }
            
            .zc-spinner--small {
                width: 1rem;
                height: 1rem;
                border-width: 0.2rem;
            }
            
            .zc-spinner--large {
                width: 4rem;
                height: 4rem;
                border-width: 0.35rem;
            }
            
            .zc-spinner--dark {
                border-color: #495057;
                border-top-color: #343a40;
            }
            
            @keyframes spin {
                from {
                    transform: rotate(0deg);
                }
                to {
                    transform: rotate(360deg);
                }
            }
            
            @media (max-width: 768px) {
                .zc-spinner {
                    width: 1.5rem;
                    height: 1.5rem;
                    border-width: 0.2rem;
                }
            }
